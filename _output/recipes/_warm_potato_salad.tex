{Warm Potato Salad}
\ingredients[9]{%
\faSquareO \ \textit{Bacon} \small{(6 slices)} \\
\faSquareO \ \textit{Mixed Baby Potatoes} \small{(\weightLbTog{3} g)} \\
\faSquareO \ \textit{Olive Oil} \small{(3 tbsp)} \\
\faSquareO \ \textit{Dijon Mustard} \small{(2 tbsp)} \\
\faSquareO \ \textit{Red Wine Vinegar} \small{(1 tbsp)} \\
\faSquareO \ \textit{Onion} \small{(1)} \\
\faSquareO \ \textit{Fresh Chives} \small{(2 tbsp)} \\
\faSquareO \ \textit{Fresh Dill} \small{(3 tbsp)} \\
\faSquareO \ \textit{Boiled Eggs} \small{(4)} \\}



\preparation{%
Chop the chives and dill, mince the onion. Halve the potatoes and eggs, and dice the bacon.}

\cooking{%
\step{Heat a skillet over medium high heat. Add bacon and cook until brown and crispy. Drain excess fat, transfer bacon to a paper towel-lined plate.}
\step{Place potatoes in a medium saucepan and cover with cold salted water by 1 inch. Bring to a boil and simmer until just tender.}
\step{In a large bowl, whisk together olive oil, Dijon, red wine vinegar, shallot, chives and dill, season with salt and pepper.}
\step{Stir in potatoes and bacon until well combined.}
\step{Serve immediately, topped with eggs.}}
