\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[10]{min}},
bakingtime = {\unit[10]{min}},
qr = {qr/ShrimpAndBroccoliStirFry.eps},
]
{Shrimp And Broccoli Stir Fry}
\ingredients[16]{%
\multicolumn{2}{l}{\textbf{Sauce}} \\
\faSquareO \ \textit{Soy Sauce} \small{(3 tbsp)} \\
\faSquareO \ \textit{Oyster Sauce} \small{(2 tbsp)} \\
\faSquareO \ \textit{Rice Wine Vinegar} \small{(1 tbsp)} \\
\faSquareO \ \textit{Brown Sugar} \small{(1 tbsp)} \\
\faSquareO \ \textit{Ginger} \small{(1 tbsp)} \\
\faSquareO \ \textit{Garlic} \small{(2 cloves)} \\
\faSquareO \ \textit{Sesame Oil} \small{(1 tsp)} \\
\faSquareO \ \textit{Cornstarch} \small{(1 tsp)} \\
\faSquareO \ \textit{Sriracha} \small{(1 tsp)} \\
\multicolumn{2}{l}{\textbf{Stir Fry}} \\
\faSquareO \ \textit{Olive Oil} \small{(1 tbsp)} \\
\faSquareO \ \textit{Shrimp} \small{(\weightLbTog{1.5} g)} \\
\faSquareO \ \textit{Broccoli Florets} \small{(\weightOzTog{24} g)} \\
\faSquareO \ \textit{Green Onion} \small{(1)} \\
\faSquareO \ \textit{Sesame Seeds} \small{(1 tsp)} \\}



\preparation{%
Thinly slice the green onion, grate the ginger and mince the garlic.}

\cooking{%
\step{In a small bowl, whisk together soy sauce, oyster sauce, rice wine vinegar, brown sugar, ginger, garlic, sesame oil, cornstarch and Sriracha.}
\step{Heat olive oil in a skillet over medium high heat. Add shrimp, and cook, stirring occasionally, until pink.}
\step{Add broccoli, and cook, stirring frequently, until tender.}
\step{Stir in soy sauce mixture until well combined and slightly thickened.}
\step{Serve immediately, garnished with green onion and sesame seeds.}}

    

    
    
\end{recipe}
