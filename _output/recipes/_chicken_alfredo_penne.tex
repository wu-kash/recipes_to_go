\begin{recipe}
[% 
portion = {\portion{6}},
preparationtime = {\unit[15]{min}},
bakingtime = {\unit[20]{min}},
qr = {qr/ChickenAlfredoPenne.eps},
]
{Chicken Alfredo Penne}
\ingredients[14]{%
\faSquareO \ \textit{Unsalted Butter} \small{(2 tbsp)} \\
\faSquareO \ \textit{Chicken Breasts} \small{(\weightLbTog{1.5} g)} \\
\faSquareO \ \textit{Dried Oregano} \small{(\sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Dried Basil} \small{(\sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Ground Black Pepper} \small{(\sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Pasta Penne} \small{(\weightOzTog{16} g)} \\
\faSquareO \ \textit{Parsley} \small{(\sfrac{1}{4} cup)} \\
\faSquareO \ \textit{Parmesan Cheese} \small{(\sfrac{1}{2} cup)} \\
\multicolumn{2}{l}{\textbf{Sauce}} \\
\faSquareO \ \textit{Garlic} \small{(4 cloves)} \\
\faSquareO \ \textit{Flour} \small{(3 tbsp)} \\
\faSquareO \ \textit{Milk} \small{(2 cups)} \\
\faSquareO \ \textit{Salt} \small{(\sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Ground Pepper} \small{(\sfrac{1}{2} tsp)} \\}



\preparation{%
Shred the parmesan cheese and chop the parsley. Mince the garlic.}

\cooking{%
\step{Cook according to the instructions and set aside.}
\step{Melt the butter in a large pan over medium-high heat, then add the chicken breasts. Season with the salt, pepper, oregano, and basil. Cook 8-10 minutes, or until the chicken is fully cooked. Remove the chicken from the pan and set aside.}
\step{In the same pan over medium heat, melt the butter, then add the garlic. Cook until the garlic begins to soften. Stir in half of the flour at a time until incorporated. Gradually pour in the milk, stirring between additions, and cook until fully incorporated and the sauce begins to thicken.}
\step{Season with the salt, pepper, oregano, and basil. Add the Parmesan cheese and stir until melted.}
\step{In a large bowl, pour the sauce over cooked penne pasta, then add the chicken and toss well to combine. Add the parsley and parmesan and toss again.}}

    

    
    
\end{recipe}
