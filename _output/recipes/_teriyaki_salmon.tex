\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[10]{min}},
bakingtime = {\unit[20]{min}},
qr = {qr/TeriyakiSalmon.eps},
]
{Teriyaki Salmon}
\ingredients[7]{%
\faSquareO \ \textit{Salmon Fillets} \small{(4)} \\
\faSquareO \ \textit{Green Beans} \small{(\weightOzTog{16} g)} \\
\faSquareO \ \textit{Carrots} \small{(2)} \\
\faSquareO \ \textit{Teriyaki Sauce} \small{(1/2 cup)} \\
\faSquareO \ \textit{Olive Oil} \small{(2 tbsp)} \\
\faSquareO \ \textit{Green Onion} \small{(2)} \\
\faSquareO \ \textit{Sesame Seeds} \small{(\sfrac{1}{2} tsp)} \\}



\preparation{%
Thinly slice the green onion, peel and cut the carrots into slices.}

\cooking{%
\step{Preheat oven to \tempFToC{400}. Lightly oil a baking sheet or coat with nonstick spray.}
\step{Place salmon, green beans and carrots in a single layer onto the prepared baking sheet.}
\step{Spoon teriyaki sauce over the salmon.}
\step{Drizzle green beans and carrots with olive oil, season with salt and pepper to taste.}
\step{Place into oven and cook until the fish flakes easily with a fork, about 15 minutes.}
\step{Serve immediately, garnished with green onions and sesame seeds.}}

    

    
    
\end{recipe}
