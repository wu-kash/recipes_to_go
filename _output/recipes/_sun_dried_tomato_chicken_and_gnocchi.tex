\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[20]{min}},
bakingtime = {\unit[40]{min}},
qr = {qr/SunDriedTomatoChickenAndGnocchi.eps},
]
{Sun Dried Tomato Chicken And Gnocchi}
\ingredients[14]{%
\faSquareO \ \textit{Gnocchi} \small{(\weightOzTog{16} g)} \\
\faSquareO \ \textit{Chicken Thighs} \small{(\weightLbTog{1.5} g)} \\
\faSquareO \ \textit{Unsalted Butter} \small{(3 tbsp)} \\
\faSquareO \ \textit{Onion} \small{(1)} \\
\faSquareO \ \textit{Sun Dried Tomatoes} \small{(\sfrac{1}{2} cup)} \\
\faSquareO \ \textit{Tomato Paste} \small{(1 tbsp)} \\
\faSquareO \ \textit{Garlic} \small{(3 cloves)} \\
\faSquareO \ \textit{Thyme Leaves} \small{(2 tsp)} \\
\faSquareO \ \textit{Flour} \small{(3 tbsp)} \\
\faSquareO \ \textit{Chicken Broth} \small{(1 \sfrac{1}{2} cups)} \\
\faSquareO \ \textit{Dry White Wine} \small{(\sfrac{1}{4} cup)} \\
\faSquareO \ \textit{Half and Half} \small{(\sfrac{1}{2} cup)} \\
\faSquareO \ \textit{Kale} \small{(1 bunch)} \\
\faSquareO \ \textit{Parmesan} \small{(\sfrac{1}{2} cup)} \\}



\preparation{%
Dice the onion, chop the thyme leaves, mince the garlic and grate the Parmesan.}

\cooking{%
\step{Preheat oven to \tempFToC{400}.}
\step{In a large pot of boiling salted water, cook gnocchi according to package instructions.}
\step{Season chicken with \sfrac{3}{4} tsp salt and \sfrac{1}{2} tsp pepper.}
\step{Melt 2 tbsp butter in a large skillet over medium heat. Add chicken to the skillet in a single layer and cook until golden brown and cooked through, set aside.}
\step{Melt remaining 1 tbsp butter. Add shallot, and cook, stirring frequently, until tender.}
\step{Stir in sun dried tomatoes, tomato paste, garlic and thyme, until fragrant, about 1 minute.}
\step{Whisk in flour until lightly browned, about 1 minute.}
\step{Gradually whisk in chicken broth and wine, whisking constantly, until slightly thickened. Stir in half and half until thickened, season with salt and pepper to taste.}
\step{Stir in gnocchi and kale until the kale has wilted. Return chicken to the skillet and sprinkle with Parmesan.}
\step{Place into oven and bake until bubbly. Serve immediately.}}

    

    
    
\end{recipe}
