\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[10]{min}},
bakingtime = {\unit[20]{min}},
]
{\includegraphics[width=1cm]{_qr/_chicken_tikka_masala.png} Chicken Tikka Masala}
\ingredients[14]{%
\faSquareO \ \textit{Rice} \small{(1 cup)} \\
\faSquareO \ \textit{Chicken Thighs} \small{(\weightLbTog{1.5} g)} \\
\faSquareO \ \textit{Oil} \small{(1 \sfrac{1}{2} tbsp)} \\
\faSquareO \ \textit{Sweet Onion} \small{(\sfrac{1}{2})} \\
\faSquareO \ \textit{Tomato Paste} \small{(3 tbsp)} \\
\faSquareO \ \textit{Garlic} \small{(3 cloves)} \\
\faSquareO \ \textit{Ginger Root} \small{(1 tbsp)} \\
\faSquareO \ \textit{Garam Masala} \small{(1 \sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Chill Powder} \small{(1 \sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Ground Tumeric} \small{(1 \sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Tomato Sauce} \small{(1 can)} \\
\faSquareO \ \textit{Chicken Stock} \small{(1 cup)} \\
\faSquareO \ \textit{Heavy Cream} \small{(\sfrac{1}{2} cup)} \\
\faSquareO \ \textit{Cilantro Leaves} \small{(2 tbsp)} \\
}



\preparation{%
Cut the chicken thighs into 2 cm chunks. Mince the garlic, grate the ginger and chop up the cilantro leaves.}

\cooking{%
\step{In a large saucepan of 2 cups water, cook rice according to package instructions and set aside.}
\step{Season chicken with 1 teaspoon salt and \sfrac{1}{2} teaspoon pepper.}
\step{Heat oil in a large stockpot over medium heat. Add chicken and onion to the stockpot and cook until golden, about 4-5 minutes.}
\step{Stir in tomato paste, garlic, ginger, garam masala, chili powder and turmeric until fragrant, about 1 minute.}
\step{Stir in tomato sauce and chicken stock and season with salt and pepper. Bring to a boil, reduce heat and simmer until reduced and slightly thickened, about 10 minutes.}
\step{Stir in heavy cream until heated through, about 1 minute.}
\step{Serve with rice and garnish with cilantro.}}

    

    
    
\end{recipe}
