\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[10]{min}},
bakingtime = {\unit[20]{min}},
]
{Chicken Tikka Masala}
    
\ingredients{%
\faSquareO \ \textit{Rice} (1 cup)  \\
\faSquareO \ \textit{Chicken Thighs} (\weightLb{1.5})  \\
\faSquareO \ \textit{Oil} (1 1/2 tbsp)  \\
\faSquareO \ \textit{Sweet Onion} (1/2)  \\
\faSquareO \ \textit{Tomato Paste} (3 tbsp)  \\
\faSquareO \ \textit{Garlic} (3 cloves)  \\
\faSquareO \ \textit{Ginger Root} (1 tbsp)  \\
\faSquareO \ \textit{Garam Masala} (1 1/2 tsp)  \\
\faSquareO \ \textit{Chill Powder} (1 1/2 tsp)  \\
\faSquareO \ \textit{Ground Tumeric} (1 1/2 tsp)  \\
\faSquareO \ \textit{Tomato Sauce} (1 can)  \\
\faSquareO \ \textit{Chicken Stock} (1 cup)  \\
\faSquareO \ \textit{Heavy Cream} (1/2 cup)  \\
\faSquareO \ \textit{Cilantro Leaves} (2 tbsp)  \\
}


\preparation{%
\\
Cut the chicken thighs into 2 cm chunks. Mince the garlic, grate the ginger and chop up the cilantro leaves.}

\cooking{%
\step In a large saucepan of 2 cups water, cook rice according to package instructions and set aside.
\step Season chicken with 1 teaspoon salt and 1/2 teaspoon pepper.
\step Heat oil in a large stockpot over medium heat. Add chicken and onion to the stockpot and cook until golden, about 4-5 minutes.
\step Stir in tomato paste, garlic, ginger, garam masala, chili powder and turmeric until fragrant, about 1 minute.
\step Stir in tomato sauce and chicken stock and season with salt and pepper. Bring to a boil, reduce heat and simmer until reduced and slightly thickened, about 10 minutes.
\step Stir in heavy cream until heated through, about 1 minute.
\step Serve with rice and garnish with cilantro.}

    

    
    
\end{recipe}
