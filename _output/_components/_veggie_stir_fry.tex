\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[10]{min}},
bakingtime = {\unit[10]{min}},
]
{Veggie Stir Fry}
    
\ingredients{%
\faSquareO \ \textit{Spring Onion} (1 bunch)  \\
\faSquareO \ \textit{Sweet Pepper} (1)  \\
\faSquareO \ \textit{Mange Tout} (100 g)  \\
\faSquareO \ \textit{Bean Sprouts} (250 g)  \\
\faSquareO \ \textit{Garlic} (1 clove)  \\
\faSquareO \ \textit{Olive Oil} (2 tbsp)  \\
\faSquareO \ \textit{Soy Sauce} (2 tbsp)  \\
\faSquareO \ \textit{Noodles/Rice} (270 g)  \\
}


\preparation{%
\\
Prepare the vegetables by cleaning and chopping the vegetables into smaller pieces.}

\cooking{%
\step Put the oil into a pan on moderate heat and crush the garlic into the pan. Stir it around.
\step Add the spring onion, sweet pepper and mange tout. Fry and stir for about 30 seconds.
\step Add the bean sprouts and soy sauce. Stir fry for about 2 minutes.
\step Serve with cooked noodles.}

    

    
    
\end{recipe}
