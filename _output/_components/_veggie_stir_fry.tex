\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[10]{min}},
bakingtime = {\unit[10]{min}},
]
{\includegraphics[width=1cm]{_qr/_veggie_stir_fry.png} Veggie Stir Fry}
\ingredients[8]{%
\faSquareO \ \textit{Spring Onion} \small{(1 bunch)} \\
\faSquareO \ \textit{Sweet Pepper} \small{(1)}  \\
\faSquareO \ \textit{Mange Tout} \small{(100 g)} \\
\faSquareO \ \textit{Bean Sprouts} \small{(250 g)} \\
\faSquareO \ \textit{Garlic} \small{(1 clove)} \\
\faSquareO \ \textit{Olive Oil} \small{(2 tbsp)} \\
\faSquareO \ \textit{Soy Sauce} \small{(2 tbsp)} \\
\faSquareO \ \textit{Noodles} \small{(270 g)} \\
}



\preparation{%
Prepare the vegetables by cleaning and chopping the vegetables into smaller pieces.}

\cooking{%
\step{Put the oil into a pan on moderate heat and crush the garlic into the pan. Stir it around.}
\step{Add the spring onion, sweet pepper and mange tout. Fry and stir for about 30 seconds.}
\step{Add the bean sprouts and soy sauce. Stir fry for about 2 minutes.}
\step{Serve with cooked noodles.}}

    

    
    
\end{recipe}
