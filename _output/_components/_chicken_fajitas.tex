\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[15]{min}},
bakingtime = {\unit[15]{min}},
]
{Chicken Fajitas}
\ingredients[14]{%
\faSquareO \ \textit{Chilli Powder} \small{(2 tsp)} \\
\faSquareO \ \textit{Ground Cumin} \small{(2 tsp)} \\
\faSquareO \ \textit{Dried Oregano} \small{(2 tsp)} \\
\faSquareO \ \textit{Smoked Paprika} \small{(1 tsp)} \\
\faSquareO \ \textit{Chicken Breasts} \small{(\weightLb{1.5})}  \\
\faSquareO \ \textit{Red Bell Pepper} \small{(1)}  \\
\faSquareO \ \textit{Yellow Bell Pepper} \small{(1)}  \\
\faSquareO \ \textit{Orange Bell Pepper} \small{(1)}  \\
\faSquareO \ \textit{Red Onion} \small{(1)}  \\
\faSquareO \ \textit{Garlic} \small{(3 cloves)} \\
\faSquareO \ \textit{Olive Oil} \small{(2 tbsp)} \\
\faSquareO \ \textit{Cilantro Leaves} \small{(\sfrac{1}{4} cup)} \\
\faSquareO \ \textit{Tortillas} \small{(4)}  \\
\faSquareO \ \textit{Lime} \small{(1)}  \\
}



\preparation{%
Cut the chicken breasts into thin strips. Cut the bell peppers and onion into slices. Mince the garlic and chop the cilantro leaves.}

\cooking{%
\step{In a small bowl, combine chili powder, cumin, oregano, paprika, and add salt and pepper, to taste. Toss the chicken strips with the mixture.}
\step{Add the olive oil over medium heat into a pan. Add the garlic and chicken strips, and cook until just cooked.}
\step{Add the onions and sliced peppers and cook for an additional 2 minutes.}
\step{Serve with tortillas and squeeze lime juice over the mix.}}

    

    
    
\end{recipe}
