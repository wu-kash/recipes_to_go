\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[10]{min}},
bakingtime = {\unit[35]{min}},
]
{Roasted Chicken With Veggies}
    
\ingredients{%
\faSquareO \ \textit{Butternut Squash} (500 g)  \\
\faSquareO \ \textit{Red Potatoes} (250 g)  \\
\faSquareO \ \textit{Carrots} (250 g)  \\
\faSquareO \ \textit{Brussels Sprouts} (250 g)  \\
\faSquareO \ \textit{Olive Oil} (2 tbsp)  \\
\faSquareO \ \textit{Balsamic Vinegar} (2 tbsp)  \\
\faSquareO \ \textit{Garlic} (2 cloves)  \\
\faSquareO \ \textit{Parsley} (2 tbsp)  \\
\faSquareO \ \textit{Chicken Thighs} (8)  \\
\faSquareO \ \textit{Oregano} (1 tsp)  \\
\faSquareO \ \textit{Thyme} (1/2 tsp)  \\
\faSquareO \ \textit{Basil} (1/2 tsp)  \\
\faSquareO \ \textit{Rosemary} (1/4 tsp)  \\
}


\preparation{%
\\
Chop the butternut squash and parsley, halve the potatoes and brussels sprouts. Mince the garlic.}

\cooking{%
\step Preheat oven to \tempF{400}. Lightly coat a baking dish with oil or nonstick spray.
\step Season chicken with oregano, thyme, basil, rosemary, salt and pepper. Set aside.
\step Place butternut squash, potatoes, carrots and brussels sprouts in a single layer onto the baking sheet. Stir in olive oil, balsamic vinegar and garlic. Season with salt and pepper and top with chicken in a single layer.
\step Place into oven and roast until the chicken is completely cooked through after about 25-30 minutes. Then broil for 2-3 minutes, or until caramelized and slightly charred.
\step Garnish with parsley and serve.}

    

    
    
\end{recipe}
