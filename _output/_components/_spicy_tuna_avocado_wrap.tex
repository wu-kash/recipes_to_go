\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[10]{min}},
]
{Spicy Tuna Avocado Wrap}
    
\ingredients{%
\faSquareO \ \textit{Tuna Can} (300)  \\
\faSquareO \ \textit{Avocado} (1)  \\
\faSquareO \ \textit{Sriracha} (1 tbsp)  \\
\faSquareO \ \textit{Dijon Mustard} (1 tbsp)  \\
\faSquareO \ \textit{Celery} (2 tbsp)  \\
\faSquareO \ \textit{Red Onion} (1 None)  \\
\faSquareO \ \textit{Green Onions} (2)  \\
\faSquareO \ \textit{Cilantro} (1 tbsp)  \\
\faSquareO \ \textit{Lettuce} (2 cups)  \\
\faSquareO \ \textit{Carrots} (1 cup)  \\
\faSquareO \ \textit{Tortillas} (4)  \\
}


\preparation{%
\\
Chop the carrots into matchsticks and the rest of the ingredients.}

\cooking{%
\step In a medium bowl, mash together tuna and avocado until combined. Add in the rest of the ingredients through the salt and pepper, mixing well.
\step To assemble, top each tortilla with a 1/2 cup leafy greens, 1/4 cup matchstick carrots and divide the tuna mixture evenly among the wraps. Tightly roll up the tortilla, slice and enjoy!}

    

    
    
\end{recipe}
