\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[10]{min}},
]
{Spicy Tuna Avocado Wrap}
\ingredients[11]{%
\faSquareO \ \textit{Tuna Can} \small{(300)}  \\
\faSquareO \ \textit{Avocado} \small{(1)}  \\
\faSquareO \ \textit{Sriracha} \small{(1 tbsp)} \\
\faSquareO \ \textit{Dijon Mustard} \small{(1 tbsp)} \\
\faSquareO \ \textit{Celery} \small{(2 tbsp)} \\
\faSquareO \ \textit{Red Onion} \small{(1 None)} \\
\faSquareO \ \textit{Green Onions} \small{(2)}  \\
\faSquareO \ \textit{Cilantro} \small{(1 tbsp)} \\
\faSquareO \ \textit{Lettuce} \small{(2 cups)} \\
\faSquareO \ \textit{Carrots} \small{(1 cup)} \\
\faSquareO \ \textit{Tortillas} \small{(4)}  \\
}



\preparation{%
Chop the carrots into matchsticks and the rest of the ingredients.}

\cooking{%
\step{In a medium bowl, mash together tuna and avocado until combined. Add in the rest of the ingredients through the salt and pepper, mixing well.}
\step{To assemble, top each tortilla with a \sfrac{1}{2} cup leafy greens, \sfrac{1}{4} cup matchstick carrots and divide the tuna mixture evenly among the wraps.}
\step{Tightly roll up the tortilla, slice and enjoy!}}

    

    
    
\end{recipe}
