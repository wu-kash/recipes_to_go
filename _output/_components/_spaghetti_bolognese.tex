\begin{recipe}
[% 
portion = {\portion{3-4}},
preparationtime = {\unit[5]{min}},
bakingtime = {\unit[30]{min}},
]
{\includegraphics[width=1cm]{_qr/_spaghetti_bolognese.png} Spaghetti Bolognese}
\ingredients[13]{%
\faSquareO \ \textit{Onions} \small{(3)}  \\
\faSquareO \ \textit{Garlic} \small{(4 cloves)} \\
\faSquareO \ \textit{Olive Oil} \small{(1 tbsp)} \\
\faSquareO \ \textit{Minced Beef} \small{(500 g)} \\
\faSquareO \ \textit{Tomatoes Can} \small{(400 g)} \\
\faSquareO \ \textit{Tomato Pur\'ee} \small{(2 tbsp)} \\
\faSquareO \ \textit{Dried Oregano} \small{(2 tsp)} \\
\faSquareO \ \textit{Bay Leaves} \small{(2)}  \\
\faSquareO \ \textit{Sugar} \small{(1 \sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Vegetable Bouillon} \small{(1 tbsp)} \\
\faSquareO \ \textit{Spaghetti} \small{(500 g)} \\
\faSquareO \ \textit{Salt} \small{(1 tsp)} \\
\faSquareO \ \textit{Pepper} \small{(1 tsp)} \\
}



\preparation{%
Peel and chop the onions and garlic.}

\cooking{%
\step{Heat the oil in a large pan and add onions and garlic. Cook on low for 2-3 minutes.}
\step{Add the minced beef and stir. Cook for 3-5 minutes until the meat has slightly browned.}
\step{Add the can of tomatoes to the pan, mash the tomatoes in the pan. Cook for 2-3 minutes and then add the tomato pur\'ee.}
\step{Add oregano, bay leaves, sugar and the vegetable bouillion and then add \sfrac{1}{2} cup of water. If the sauce gets too dry, add some more water.}
\step{Add pepper, to taste and cook on low heat for about 10-15 minutes.}
\step{Boil salted water, add spaghetti and follow the cooking instructions.}
\step{Serve the pasta and cover it with the bolognese sauce.}}

    

    
    
\end{recipe}
