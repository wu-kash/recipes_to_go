\begin{recipe}
[% 
portion = {\portion{6}},
preparationtime = {\unit[30]{min}},
bakingtime = {None},
]
{\includegraphics[width=1cm]{_qr/_bean_broccoli_salad.png} Bean Broccoli Salad}
\ingredients[11]{%
\multicolumn{2}{l}{\textbf{Salad Dressing}} \\
\faSquareO \ \textit{Parsley} \small{(\sfrac{1}{2} cup)} \\
\faSquareO \ \textit{Red Onion} \small{(1)} \\
\faSquareO \ \textit{Rucola} \small{(4 cups)} \\
\faSquareO \ \textit{Olive Oil} \small{(1 \sfrac{1}{2} tbsp)} \\
\faSquareO \ \textit{Salt} \small{(\sfrac{1}{2} tsp)} \\
\multicolumn{2}{l}{\textbf{Salad}} \\
\faSquareO \ \textit{Broccoli Florets} \small{(450 g)} \\
\faSquareO \ \textit{Pepper} \small{(\sfrac{1}{4} tsp)} \\
\faSquareO \ \textit{White Beans} \small{(400 g)} \\
\faSquareO \ \textit{Parmessan} \small{(\sfrac{1}{2} cup)} \\
}



\preparation{%
Chop the parsley and mince the garlic. Drain the beans from the cans. Thinly slice the red onion and grate the parmesan cheese.}

\cooking{%
\step{Preheat oven to \tempFToC{400} with the oven rack 10 cm from the heat. Toss together broccoli, 1 \sfrac{1}{2} tbsp of the oil, \sfrac{1}{2} tsp of the salt, and \sfrac{1}{4} teaspoon of the pepper in a large bowl. Spread broccoli in a single layer on a rimmed baking sheet. Roast in oven until just tender, 20 to 25 minutes.}
\step{Whisk together lemon zest and juice, parsley, garlic, mustard, vinegar, honey, and remaining \sfrac{1}{2} cup oil, \sfrac{1}{2} tsp salt, and \sfrac{1}{4} tsp pepper in a medium bowl. Set aside.}
\step{Increase oven temperature to broil, cook broccoli until slightly charred, 1 to 2 minutes. Cool about 5 minutes.}
\step{Combine broccoli, beans, rucola, and red onion in a large bowl. Toss with the dressing and sprinkle with parmesan.}}

    

    
    
\end{recipe}
