\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[15]{min}},
bakingtime = {\unit[10]{min}},
]
{Korean Beef Zucchini Noodles}
    
\ingredients{%
\faSquareO \ \textit{Sugar} (1/3 cup)  \\
\faSquareO \ \textit{Soy Sauce} (1 tbsp)  \\
\faSquareO \ \textit{Ginger} (1 tbsp)  \\
\faSquareO \ \textit{Sesame Oil} (1 tbsp)  \\
\faSquareO \ \textit{Sriracha} (1 tsp)  \\
\faSquareO \ \textit{Olive Oil} (1 tbsp)  \\
\faSquareO \ \textit{Garlic} (1 tsp)  \\
\faSquareO \ \textit{Ground Beef} (\weightLb{1})  \\
\faSquareO \ \textit{Zucchini} (\weightLb{1.5})  \\
\faSquareO \ \textit{Green Onion} (2)  \\
\faSquareO \ \textit{Sesame Seeds} (1/4 tsp)  \\
}


\preparation{%
\\
Grate the ginger, mince the garlic and thinly slice the green onions.}

\cooking{%
\step In a small bowl, whisk together brown sugar, soy sauce, ginger, sesame oil and Sriracha.
\step Heat olive oil in a large skillet over medium high heat. Add garlic and cook, stirring constantly, until fragrant, about 1 minute. Add ground beef and cook until browned, about 3-5 minutes, making sure to crumble the beef as it cooks, drain excess fat.
\step Stir in zucchini noodles, green onions and soy sauce mixture until well combined, allowing to simmer until heated through, about 2 minutes.
\step Garnish with sesame seeds and serve.}

    

    
    
\end{recipe}
