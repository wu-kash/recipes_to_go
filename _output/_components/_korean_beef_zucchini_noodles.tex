\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[15]{min}},
bakingtime = {\unit[10]{min}},
]
{Korean Beef Zucchini Noodles}
\ingredients[11]{%
\faSquareO \ \textit{Sugar} \small{(\sfrac{1}{3} cup)} \\
\faSquareO \ \textit{Soy Sauce} \small{(1 tbsp)} \\
\faSquareO \ \textit{Ginger} \small{(1 tbsp)} \\
\faSquareO \ \textit{Sesame Oil} \small{(1 tbsp)} \\
\faSquareO \ \textit{Sriracha} \small{(1 tsp)} \\
\faSquareO \ \textit{Olive Oil} \small{(1 tbsp)} \\
\faSquareO \ \textit{Garlic} \small{(1 tsp)} \\
\faSquareO \ \textit{Ground Beef} \small{(\weightLb{1})}  \\
\faSquareO \ \textit{Zucchini} \small{(\weightLb{1.5})}  \\
\faSquareO \ \textit{Green Onion} \small{(2)}  \\
\faSquareO \ \textit{Sesame Seeds} \small{(\sfrac{1}{4} tsp)} \\
}



\preparation{%
Grate the ginger, mince the garlic and thinly slice the green onions.}

\cooking{%
\step{In a small bowl, whisk together brown sugar, soy sauce, ginger, sesame oil and Sriracha.}
\step{Heat olive oil in a large skillet over medium high heat. Add garlic and cook, stirring constantly, until fragrant, about 1 minute. Add ground beef and cook until browned, making sure to crumble the beef as it cooks.}
\step{Stir in zucchini noodles, green onions and soy sauce mixture until well combined, allowing to simmer until heated through, about 2 minutes.}
\step{Garnish with sesame seeds and serve.}}

    

    
    
\end{recipe}
