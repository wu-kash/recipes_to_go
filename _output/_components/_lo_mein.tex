\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[15]{min}},
bakingtime = {\unit[5]{min}},
]
{\includegraphics[width=1cm]{_qr/_lo_mein.png} Lo Mein}
\ingredients[10]{%
\faSquareO \ \textit{Dark Soy Sauce} \small{(2 tbsp)} \\
\faSquareO \ \textit{Light Soy Sauce} \small{(1 tbsp)} \\
\faSquareO \ \textit{Oyster Sauce} \small{(3 tbsp)} \\
\faSquareO \ \textit{Sesame Oil} \small{(1 tbsp)} \\
\faSquareO \ \textit{Sugar} \small{(1 tsp)} \\
\faSquareO \ \textit{Garlic Clove} \small{(1)}  \\
\faSquareO \ \textit{Ramen Noodles} \small{(175 g)} \\
\faSquareO \ \textit{Green Onion} \small{(3)}  \\
\faSquareO \ \textit{Mixed Veggies} \small{(200 g)} \\
\faSquareO \ \textit{Rice Vinegar} \small{(1 \sfrac{1}{2} tbsp)} \\
}



\preparation{%
Chop the grreen onion and mince the garlic. Julienne cut or chop the mixed vegetables (carrots, mushrooms, broccoli, red peppers, cabbage)}

\cooking{%
\step{Shake all the sauce ingredients together in a jar.}
\step{Cook the noodles according to package directions. Drain and set aside.}
\step{Heat the sesame oil in a large wok or skillet. Add the green onions (white parts) and vegetables to the hot pan. Stir fry until fork-tender, about 5 minutes.}
\step{Add the rice vinegar to loosen the browned bits up off the bottom of the pan. Add the cooked noodles and about half of the sauce, toss around in the hot pan to combine. Add more sauce if needed.}
\step{Garnish with remaining green onions and serve.}}

    

    
    
\end{recipe}
