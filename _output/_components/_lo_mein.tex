\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[15]{min}},
bakingtime = {\unit[5]{min}},
]
{Lo Mein}
    
\ingredients{%
\faSquareO \ \textit{Dark Soy Sauce} (2 tbsp)  \\
\faSquareO \ \textit{Light Soy Sauce} (1 tbsp)  \\
\faSquareO \ \textit{Oyster Sauce} (3 tbsp)  \\
\faSquareO \ \textit{Sesame Oil} (1 tbsp)  \\
\faSquareO \ \textit{Sugar} (1 tsp)  \\
\faSquareO \ \textit{Garlic Clove} (1)  \\
\faSquareO \ \textit{Ramen Noodles} (175 g)  \\
\faSquareO \ \textit{Green Onion} (3)  \\
\faSquareO \ \textit{Mixed Veggies} (200 g)  \\
\faSquareO \ \textit{Mirrin} (1 1/2 tbsp)  \\
}


\preparation{%
\\
Chop the grreen onion and mince the garlic. Julienne cut or chop the mixed vegetables (carrots, mushrooms, broccoli, red peppers, cabbage)}

\cooking{%
\step Shake all the sauce ingredients together in a jar.
\step Cook the noodles according to package directions. Drain and set aside.
\step Heat the sesame oil in a large wok or skillet. Add the green onions (white parts) and vegetables to the hot pan. Stir fry until fork-tender, about 5 minutes. Add the mirin to loosen the browned bits up off the bottom of the pan. Add the cooked noodles and about half of the sauce, toss around in the hot pan to combine. Add more sauce if needed (want a medium brown color, not too light, not too dark). Serve topped with remaining green onions!}

    

    
    
\end{recipe}
