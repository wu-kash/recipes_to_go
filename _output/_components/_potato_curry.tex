\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[15]{min}},
bakingtime = {\unit[25]{min}},
]
{\includegraphics[width=1cm]{_qr/_potato_curry.png} Potato Curry}
\ingredients[10]{%
\faSquareO \ \textit{Potatoes} \small{(400 g)} \\
\faSquareO \ \textit{Garlic} \small{(1 clove)} \\
\faSquareO \ \textit{Ginger Root} \small{(1 cm)} \\
\faSquareO \ \textit{Onion} \small{(1)} \\
\faSquareO \ \textit{Sunflower Oil} \small{(3 tbsp)} \\
\faSquareO \ \textit{Cumin Seeds} \small{(1 tsp)} \\
\faSquareO \ \textit{Cardamom Pod} \small{(1)} \\
\faSquareO \ \textit{Chilli Powder} \small{(\sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Brown Mustard Seeds} \small{(2 tbsp)} \\
\faSquareO \ \textit{Garam Masala} \small{(1 tsp)} \\
}



\preparation{%
Clean the potatoes, cut the onion into slices. Peel and crush the gartlic and ginger root}

\cooking{%
\step{Boil the potatoes in salted water for about 15 minutes until they are just done. Drain and cut in half.}
\step{Put the oil into a frying pan over moderate heat. Fry the cumin, cardamom, chilli powder and mustard seeds for about 1 minute.}
\step{Add the garlic, ginger and onion and fry for about 3 minutes until the onion is golden. Keep stirring.}
\step{Add the potatoes and garam masala, fry for about 5 minutes, stirring.}
\step{Can be served with rice.}}

    

    
    
\end{recipe}
