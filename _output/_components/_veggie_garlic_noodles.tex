\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[5]{min}},
bakingtime = {\unit[10]{min}},
]
{Veggie Garlic Noodles}
    
\ingredients{%
\faSquareO \ \textit{Rice Noodles} (250 g)  \\
\faSquareO \ \textit{Soy Sauce} (3 tbsp)  \\
\faSquareO \ \textit{Brown Sugar} (2 tbsp)  \\
\faSquareO \ \textit{Carrots} (2)  \\
\faSquareO \ \textit{Snap Peas} (100 g)  \\
\faSquareO \ \textit{Green Onion} (4)  \\
\faSquareO \ \textit{Garlic} (5 cloves)  \\
}


\preparation{%
\\
Cut the carrots into matchsticks, slice the green onion and mince the garlic.}

\cooking{%
\step Cook the rice noodles according to he instructions.
\step Heat the vegetable oil in a large skillet over medium heat. Add the garlic and 3 green onions and cook for 2-3 minutes, until fragrant and tender. Add the carrots and snap peas, then stir in the brown sugar and soy sauce.
\step Add the noodles and toss for a couple of minutes, until the noodles are well-coated and the vegetables are tender. Season with salt and pepper.
\step Serve immediately, garnished with the remaining green onion.}

    

    
    
\end{recipe}
