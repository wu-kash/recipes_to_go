\begin{recipe}
[% 
portion = {\portion{6}},
preparationtime = {\unit[10]{min}},
bakingtime = {\unit[40]{min}},
]
{Garlic Parmesan Chicken And Potatoes}
\ingredients[16]{%
\multicolumn{2}{l}{\textbf{Main Dish}} \\
\faSquareO \ \textit{Chicken Thighs} \small{(6)}  \\
\faSquareO \ \textit{Italian Herbs} \small{(1 tbsp)} \\
\faSquareO \ \textit{Unsalted Butter} \small{(\sfrac{1}{4} cup)} \\
\faSquareO \ \textit{Baby Spinach} \small{(3 cups)} \\
\faSquareO \ \textit{Potatoes} \small{(500 g)} \\
\faSquareO \ \textit{Parsley} \small{(2 tbsp)} \\
\multicolumn{2}{l}{\textbf{Garlic Parmesan Sauce}} \\
\faSquareO \ \textit{Garlic} \small{(4 cloves)} \\
\faSquareO \ \textit{Flour} \small{(2 tbsp)} \\
\faSquareO \ \textit{Chicken Broth} \small{(1 cup)} \\
\faSquareO \ \textit{Dried Thyme} \small{(1 tsp)} \\
\faSquareO \ \textit{Dried Basil} \small{(\sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Milk} \small{(\sfrac{1}{2} cup)} \\
\faSquareO \ \textit{Cream} \small{(\sfrac{1}{2} cup)} \\
\faSquareO \ \textit{Parmesan} \small{(\sfrac{1}{2} cup)} \\
}



\preparation{%
Mince garlic, grate parmesan.
Chop the baby spinach and parsley leaves.
Halve the potatoes.}

\cooking{%
\step{Preheat oven to \tempF{400}. Lightly coat a baking dish with oil or nonstick spray.}
\step{Season chicken with the Italian herbs, salt and pepper.}
\step{Melt 2 tablespoons butter in a large skillet over medium high heat. Add the chicken thighs and sear until both sides are golden brown. Set aside.}
\step{Melt the remaining 1 tablespoon of butter in the skillet. Stir in spinach and cook, stirring occassionally until it begins to wilt.}
\step{For the garlic parmesan sauce, melt the butter in the skillet over medium heat. Add the garlic and cooking for about 2 minutes, stirring frequently.}
\step{Whisk in the flour until lightly browned, about 1 minute.}
\step{Gradually whisk in chicken broth, thyme and basil. Cook for about 2 minutes. Stir in the milk, cream and the parmesan until the mix has slightly thickened. Add more milk if the mixture is too thick.}
\step{Place the chicken in a single layer into the baking dish. Add the potatoes on top, the spinach and the cream sauce.}
\step{Roast in the oven until cooked through, about 25-30 minutes.}}

    

    
    
\end{recipe}
