\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[15]{min}},
bakingtime = {\unit[15]{min}},
]
{Lemon Garlic Shrimp Pasta}
    
\ingredients{%
\faSquareO \ \textit{Linguine} (250 g)  \\
\faSquareO \ \textit{Olive Oil} (2 tbsp)  \\
\faSquareO \ \textit{Unsalted Butter} (8 tbsp)  \\
\faSquareO \ \textit{Garlic} (4 cloves)  \\
\faSquareO \ \textit{Repper Pepper Flakes} (1 tsp)  \\
\faSquareO \ \textit{Shrimp} (500 g)  \\
\faSquareO \ \textit{Dried Oregano} (1 tsp)  \\
\faSquareO \ \textit{Baby Spinach} (4 cups)  \\
\faSquareO \ \textit{Parmesan Cheese} (1/4 cup)  \\
\faSquareO \ \textit{Parsley} (2 tbsp)  \\
\faSquareO \ \textit{Lemon Juice} (1 tbsp)  \\
\faSquareO \ \textit{Pepper} (1 tsp)  \\
\faSquareO \ \textit{Salt} (1 tsp)  \\
}


\preparation{%
\\
Mince the garlic, grate the parmesan cheese, and chop the parsley.}

\cooking{%
\step In a large pot, boil water and add pasta. Cook until al dente.
\step Drain and set pasta aside.
\step In the same pan, heat olive oil and 2 tbsp of butter. Add garlic and crushed red pepper, cook until fragrant.
\step Toss in shrimp, add salt and pepper, to taste, and stir until shrimp start to turn pink, but are not fully cooked.
\step Add oregano and spinach, cook until wilted.
\step Return cooked pasta to the pot, add remaining butter, parmesan, and parsley. Stir until well mixed and the butter is melted.
\step When the shrimp are cooked, add lemon juice, mix once more, then serve while hot.}

    

    
    
\end{recipe}
