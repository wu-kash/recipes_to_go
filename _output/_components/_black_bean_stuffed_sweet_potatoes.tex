\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[15]{min}},
bakingtime = {\unit[60]{min}},
]
{Black Bean Stuffed Sweet Potatoes}
    
\ingredients{%
\faSquareO \ \textit{Sweet Potato} (4)  \\
\faSquareO \ \textit{Black Beans} (450 g)  \\
\faSquareO \ \textit{Plain Greek Yoghurt} (1/2 cup)  \\
\faSquareO \ \textit{Lime Juice} (1 tsp)  \\
\faSquareO \ \textit{Ground Black Pepper} (1/2 tsp)  \\
\faSquareO \ \textit{Red Onion} (1 cup)  \\
\faSquareO \ \textit{Olive Oil} (1 1/2 tbsp)  \\
\faSquareO \ \textit{Garlic Powder} (1/4 tsp)  \\
\faSquareO \ \textit{Onion Powder} (1/4 tsp)  \\
\faSquareO \ \textit{Chilli Powder} (1/4 tsp)  \\
\faSquareO \ \textit{Cumin} (1/4 tsp)  \\
\faSquareO \ \textit{Salt} (1/2 tsp)  \\
\faSquareO \ \textit{Cilantro} (1/2 cup)  \\
\faSquareO \ \textit{Avocado} (1/2)  \\
}


\preparation{%
\\
Finely dice the red onion, chop the cilantro and avocado.}

\cooking{%
\step Pre-heat oven to \tempC{180}.
\step Place the sweet potatoes on a lightly greased baking tray and bake for about 60 minutes or until a fork can easily be inserted into the flesh of the potatoes.
\step Prepare the cream sauce by whisking together the cashew cream sauce, lime juice, and black pepper. Set aside.
\step When the sweet potatoes have about 10 minutes left, heat the oil in a skillet over medium heat and then add in onion. Saut\'e for 5 minutes until the onions begins to become translucent. Add in the spices stir and cook for an additional 3 minutes.
\step Add the black beans to the skillet and toss to combine. Continue to cook, stirring frequently, until the black beans are fully heated through. About 5 minutes. Take off of heat and set aside.
\step Once the sweet potatoes are done baking, let cool slightly. Once cool, cut the sweet potatoes open and shred the flesh from the skin so that the inside is mashed and easy to scoop out.
\step Evenly spread the black bean mixture, avocado, cilantro, and cashew crema sauce on top of the 4 sweet potatoes.}

    

    
    
\end{recipe}
