\begin{recipe}
[% 
portion = {\portion{6}},
preparationtime = {\unit[15]{min}},
bakingtime = {\unit[15]{min}},
]
{Beef Noodle Stir Fry}
    
\ingredients{%
\multicolumn{2}{l}{\textbf{Main Dish}} \\
\faSquareO \ \textit{Udon Noodles} (\weightOz{14})  \\
\faSquareO \ \textit{Olive Oil} (1 tbsp)  \\
\faSquareO \ \textit{Ground Beef} (\weightLb{1})  \\
\faSquareO \ \textit{Onion} (1)  \\
\faSquareO \ \textit{Red Bell Pepper} (1)  \\
\faSquareO \ \textit{Green Bell Pepper} (1)  \\
\faSquareO \ \textit{Broccoli Florets} (\weightOz{12})  \\
\faSquareO \ \textit{Green Onion} (1)  \\
\multicolumn{2}{l}{\textbf{Sauce}} \\
\faSquareO \ \textit{Soy Sauce} (1/4 cup)  \\
\faSquareO \ \textit{Rice Wine Vinegar} (2 tbsp)  \\
\faSquareO \ \textit{Sugar} (2 tbsp)  \\
\faSquareO \ \textit{Garlic} (3 cloves)  \\
\faSquareO \ \textit{Ginger} (1 tbsp)  \\
\faSquareO \ \textit{Sesame Oil} (1 tsp)  \\
\faSquareO \ \textit{Sriracha} (1 tsp)  \\
}


\preparation{%
\\
Dice the onion. Thinly slice the bell peppers and green onion. Grate the ginger and mince the garlic.}

\cooking{%
\step In a small bowl, whisk together soy sauce, rice wine vinegar, brown sugar, garlic, ginger, sesame oil and Sriracha. Set aside.
\step In a large pot of boiling water, cook udon noodles according to package instructions and drain the water.
\step Heat olive oil in a large skillet over medium high heat. Add ground beef and onion and cook until beef has browned, about 3-5 minutes, making sure to crumble the beef as it cooks and drain excess fat.
\step Stir in bell peppers and broccoli. Cook, stirring frequently, until tender, about 3-4 minutes. Stir in udon noodles and soy sauce mixture until well combined, about 2-3 minutes.
\step Garnish with green onion and serve.}

    

    
    
\end{recipe}
