\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[10]{min}},
bakingtime = {\unit[20]{min}},
]
{Creamy Mushroom Stroganoff}
    
\ingredients{%
\faSquareO \ \textit{Pasta Shells} (\weightOz{8})  \\
\faSquareO \ \textit{Unsalted Butter} (3 tbsp)  \\
\faSquareO \ \textit{Mushrooms} (\weightLb{1.5})  \\
\faSquareO \ \textit{Onion} (2)  \\
\faSquareO \ \textit{Garlic} (3 cloves)  \\
\faSquareO \ \textit{Thyme} (4 tsp)  \\
\faSquareO \ \textit{Flour} (2 1/2 tbsp)  \\
\faSquareO \ \textit{Beef Stock} (2 cups)  \\
\faSquareO \ \textit{Dijon Mustard} (1 1/2 tsp)  \\
\faSquareO \ \textit{Sour Cream} (3/4 cup)  \\
\faSquareO \ \textit{Parmesan} (2/3 cup)  \\
\faSquareO \ \textit{Parsley} (2 tbsp)  \\
}


\preparation{%
\\
Thinly slice the mushrooms, dice the onions, chop the thyme, parsley and mince the garlic. Grate the parmesan.}

\cooking{%
\step In a large pot of boiling salted water, cook pasta according to package instructions and drain the water.
\step Melt butter in a large skillet over medium high heat. Add mushrooms and onions, and cook, stirring occasionally, until mushrooms are tender and browned, about 5-7 minutes. Season with salt and pepper, to taste.
\step Stir in garlic and thyme until fragrant.
\step Whisk in flour until lightly browned.
\step Gradually whisk in beef stock and Dijon. Bring to a boil, reduce heat and simmer, stirring occasionally, until reduced and slightly thickened, about 4-5 minutes.
\step Stir in pasta and sour cream until heated through. Stir in Parmesan until melted. Stir in parsley, season with salt and pepper, to taste.}

    

    
    
\end{recipe}
