\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[10]{min}},
bakingtime = {\unit[35]{min}},
]
{Meatloaf And Veggies}
\ingredients[11]{%
\faSquareO \ \textit{Ground Beef} \small{(500 g)} \\
\faSquareO \ \textit{Eggs} \small{(2)}  \\
\faSquareO \ \textit{Bread Crumbs} \small{(4 tbsp)} \\
\faSquareO \ \textit{BBQ Sauce} \small{(6 tbsp)} \\
\faSquareO \ \textit{Smoked Paprika} \small{(\sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Garlic Powder} \small{(\sfrac{1}{4} tsp)} \\
\faSquareO \ \textit{Sweet Potato} \small{(2)}  \\
\faSquareO \ \textit{Broccoli Floret} \small{(1 head)} \\
\faSquareO \ \textit{Olive Oil} \small{(4 tbsp)} \\
\faSquareO \ \textit{Salt} \small{(1 \sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Pepper} \small{(1 tsp)} \\
}



\preparation{%
Peel the sweet potato and cut it into 1 cm cubes.}

\cooking{%
\step{Preheat the oven to \tempF{400}.}
\step{Place the sweet potatoes and frozen broccoli florets (no need to thaw) on a large baking sheet.}
\step{Drizzle the olive oil over the sweet potatoes and broccoli florets. Sprinkle the seasoning salt over the sweet potatoes and then season the broccoli florets with a pinch of salt and pepper.}
\step{Toss the vegetables until they are coated in oil and spices, keeping the sweet potatoes on one side of the baking sheet and the broccoli on the other.}
\step{Transfer the baking sheet to the preheated oven and roast the vegetables for 15 minutes.}
\step{While the vegetables are roasting, prepare the meatloaves. In a bowl combine the ground beef, egg, bread crumbs, 1 tbsp of the BBQ sauce, the smoked paprika, garlic powder, and salt. Work the ingredients together with your hands until they are well combined. Divide the meatloaf mixture in two and shape each half into a flattened oval.}
\step{After the vegetables have roasted for 15 minutes, remove the baking sheet and stir each of the vegetables. Push the vegetables to the side. Place the meatloaves in the center and then spread 1 tbsp of BBQ sauce over each loaf.}
\step{Return the baking sheet to the oven and roast for an additional 20 minutes. Remove the baking sheet from the oven and serve.}}

    

    
    
\end{recipe}
