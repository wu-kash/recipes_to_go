\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[5]{min}},
bakingtime = {\unit[10]{min}},
]
{Beef Broccoli}
    
\ingredients{%
\faSquareO \ \textit{Flank Steak} (\weightLb{1})  \\
\faSquareO \ \textit{Broccoli Floret} (1 head)  \\
\faSquareO \ \textit{Chicken Stock} (1/2 cup)  \\
\faSquareO \ \textit{Soy Sauce} (2/3 cup)  \\
\faSquareO \ \textit{Honey} (1/4 cup)  \\
\faSquareO \ \textit{Rice Wine Vinegar} (2 tbsp)  \\
\faSquareO \ \textit{Brown Sugar} (2 tbsp)  \\
\faSquareO \ \textit{Garlic} (3 cloves)  \\
\faSquareO \ \textit{Sesame Oil} (1 tbsp)  \\
\faSquareO \ \textit{Olive Oil} (1 tbsp)  \\
\faSquareO \ \textit{Cornstarch} (1 tbsp)  \\
\faSquareO \ \textit{Sriracha} (1 tsp)  \\
\faSquareO \ \textit{Ground Ginger} (1 tsp)  \\
\faSquareO \ \textit{Red Pepper Flakes} (1/4 tsp)  \\
}


\preparation{%
\\
Thinly slice the flank steak and cut the broccoli head into florets. Mince the garlic.}

\cooking{%
\step In a medium bowl, whisk together soy sauce, chicken stock, honey, vinegar, brown sugar, garlic, sesame oil, cornstarch, Sriracha, ginger, red pepper flakes and 1/4 cup water; set aside.
\step Heat olive oil in a large skillet over medium high heat. Add steak and cook, flipping once, until browned, about 3-4 minutes.
\step Stir in broccoli and soy sauce mixture until tender and slightly thickened, about 3-4 minutes.}

    

    
    
\end{recipe}
