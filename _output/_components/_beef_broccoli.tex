\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[5]{min}},
bakingtime = {\unit[10]{min}},
]
{\includegraphics[width=1cm]{_qr/_beef_broccoli.png} Beef Broccoli}
\ingredients[14]{%
\faSquareO \ \textit{Flank Steak} \small{(\weightLbTog{1} g)} \\
\faSquareO \ \textit{Broccoli Floret} \small{(1 head)} \\
\faSquareO \ \textit{Chicken Stock} \small{(\sfrac{1}{2} cup)} \\
\faSquareO \ \textit{Soy Sauce} \small{(\sfrac{2}{3} cup)} \\
\faSquareO \ \textit{Honey} \small{(\sfrac{1}{4} cup)} \\
\faSquareO \ \textit{Rice Wine Vinegar} \small{(2 tbsp)} \\
\faSquareO \ \textit{Brown Sugar} \small{(2 tbsp)} \\
\faSquareO \ \textit{Garlic} \small{(3 cloves)} \\
\faSquareO \ \textit{Sesame Oil} \small{(1 tbsp)} \\
\faSquareO \ \textit{Olive Oil} \small{(1 tbsp)} \\
\faSquareO \ \textit{Cornstarch} \small{(1 tbsp)} \\
\faSquareO \ \textit{Sriracha} \small{(1 tsp)} \\
\faSquareO \ \textit{Ground Ginger} \small{(1 tsp)} \\
\faSquareO \ \textit{Red Pepper Flakes} \small{(\sfrac{1}{4} tsp)} \\
}



\preparation{%
Thinly slice the flank steak and cut the broccoli head into florets. Mince the garlic.}

\cooking{%
\step{In a medium bowl, whisk together soy sauce, chicken stock, honey, vinegar, brown sugar, garlic, sesame oil, cornstarch, Sriracha, ginger, red pepper flakes and \sfrac{1}{4} cup water; set aside.}
\step{Heat olive oil in a large skillet over medium high heat. Add steak and cook, flipping once, until browned, about 3-4 minutes.}
\step{Stir in broccoli and soy sauce mixture until tender and slightly thickened, about 3-4 minutes.}}

    

    
    
\end{recipe}
