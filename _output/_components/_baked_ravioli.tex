\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[10]{min}},
bakingtime = {\unit[20]{min}},
]
{\includegraphics[width=1cm]{_qr/_baked_ravioli.png} Baked Ravioli}
\ingredients[8]{%
\faSquareO \ \textit{Ravioli} \small{(\weightOz{16})}  \\
\faSquareO \ \textit{Marinara Sauce} \small{(2 cups)} \\
\faSquareO \ \textit{Cream Cheese} \small{(\sfrac{1}{3} cups)} \\
\faSquareO \ \textit{Dried Thyme} \small{(\sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Dried Parsley} \small{(\sfrac{1}{2} tsp)} \\
\faSquareO \ \textit{Mozzarella} \small{(\sfrac{1}{2} cup)} \\
\faSquareO \ \textit{Parmesan Cheese} \small{(\sfrac{1}{4} cup)} \\
\faSquareO \ \textit{Basil Leaves} \small{(\sfrac{1}{4} cup)} \\
}



\preparation{%
Grate the Parmesan cheese and shred the mozzarella.}

\cooking{%
\step{Preheat oven to \tempF{350}. Lightly oil a baking dish or coat with nonstick spray.}
\step{In a large pot of boiling water, cook ravioli just until tender, about 2-3 minutes.}
\step{Combine marinara sauce and cream cheese in a medium saucepan over medium heat. Cook, stirring constantly, until cream cheese has melted, about 2-3 minutes. Stir in thyme and parsley.}
\step{Spread ravioli into the prepared baking dish. Top with marinara mixture and sprinkle with mozzarella. Place into oven and bake until cheese has melted, about 20 minutes.}
\step{Serve immediately, garnished with Parmesan and basil}}

    

    
    
\end{recipe}
