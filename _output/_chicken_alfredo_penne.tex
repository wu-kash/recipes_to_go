\begin{recipe}
[% 
portion = {\portion{6}},
preparationtime = {\unit[15]{min}},
bakingtime = {\unit[20]{min}},
]
{Chicken Alfredo Penne}
    
\ingredients{%
2 tbsp & Unsalted Butter   \\
700 g & Chicken Breasts   \\
1/2 tsp & Dried Oregano   \\
1/2 tsp & Dried Basil   \\
1/2 tsp & Ground Black Pepper   \\
500 g & Pasta Penne   \\
1/4 cup & Parsley   \\
1/4 cup & Parmesan Cheese   \\}


\preparation{%
\step Melt the butter in a large pan over medium-high heat, then add the chicken breasts. Season with the salt, pepper, oregano, and basil. Cook 8-10 minutes, or until the chicken is fully cooked. Remove the chicken from the pan and set aside.
\step In the same pan over medium heat, melt the butter, then add the garlic. Cook until the garlic begins to soften. Stir in half of the flour at a time until incorporated. Gradually pour in the milk, stirring between additions, and cook until fully incorporated and the sauce begins to thicken. Season with the salt, pepper, oregano, and basil. Add the Parmesan cheese and stir until melted.
\step In a large bowl, pour the sauce over cooked penne pasta, then add the chicken and toss well to combine. Add the parsley and Parmesan and toss again.}
