\begin{recipe}
[% 
portion = {\portion{4}},
preparationtime = {\unit[15]{min}},
bakingtime = {\unit[15]{min}},
qr = {qr/CilantroLimeDressingSalad.eps},
]
{Cilantro Lime Dressing Salad}
\ingredients[14]{%
\multicolumn{2}{l}{\textbf{Cilanto Lime Dressing}} \\
\faSquareO \ \textit{Cilantro} \small{(2 cups)} \\
\faSquareO \ \textit{Plain Greek Yoghurt} \small{(1 cup)} \\
\faSquareO \ \textit{Garlic} \small{(4 cloves)} \\
\faSquareO \ \textit{Lime Juice} \small{(4 tbsp)} \\
\faSquareO \ \textit{Olive Oil} \small{(\sfrac{1}{2} cup)} \\
\faSquareO \ \textit{Apple Cider Vinegar} \small{(4 tbsp)} \\
\multicolumn{2}{l}{\textbf{Salad}} \\
\faSquareO \ \textit{Romaine Lettuce} \small{(10 cups)} \\
\faSquareO \ \textit{Cherry Tomatoes} \small{(1 cup)} \\
\faSquareO \ \textit{Canned Corn Kernels} \small{(1 cup)} \\
\faSquareO \ \textit{Canned Black Beans} \small{(1 cup)} \\
\faSquareO \ \textit{Cilantro Leaves} \small{(4 tbsp)} \\
\faSquareO \ \textit{Avocado} \small{(2)} \\}



\preparation{%
Chop the romaine lettuce, halve the cherry tomatoes. Drain the corn kernels and black beans. Remove stems from cilantro. Dice the avocado.}

\cooking{%
\step{To make the the dressing, combine the cilantro, Greek yogurt, garlic and lime juice in a food processor. While running, add olive oil and vinegar in a slow stream until emulsified.}
\step{To assemble the salad, place romaine lettuce in a large bowl; top with tomatoes, corn, black beans and cilantro. Pour the dressing on top of the salad and gently toss to combine. Stir in avocado.}}

    

    
    
\end{recipe}
